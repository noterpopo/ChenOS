nasm -f elf -o build/print.o lib/kernel/print.S
nasm -f elf -o build/kernel.o kernel/kernel.S
gcc -m32 -ffreestanding -nostdlib -mno-red-zone -I lib/kernel/  -I device/ -c device/timer.c -o build/timer.o
gcc -m32 -ffreestanding -nostdlib -mno-red-zone -I lib/kernel/  -I kernel/ -c kernel/main.c -o build/main.o
gcc -m32 -ffreestanding -nostdlib -mno-red-zone -I lib/kernel/  -I kernel/ -c kernel/init.c -o build/init.o
gcc -m32 -ffreestanding -nostdlib -mno-red-zone -I lib/kernel/  -I kernel/ -c kernel/interrupt.c -o build/interrupt.o
gcc -m32 -ffreestanding -nostdlib -mno-red-zone -I lib/kernel/  -I kernel/ -c kernel/debug.c -o build/debug.o
objcopy --remove-section .eh_frame build/main.o
objcopy --remove-section .eh_frame build/interrupt.o
objcopy --remove-section .eh_frame build/timer.o
objcopy --remove-section .eh_frame build/init.o
objcopy --remove-section .eh_frame build/debug.o
ld -m elf_i386 -Ttext 0xc0001500 -e main -o build/kernel.bin build/main.o build/init.o build/interrupt.o build/print.o build/kernel.o build/timer.o build/debug.o

dd if=build/kernel.bin of=../bochs/hd60M.img bs=512 count=200 seek=9 conv=notrunc
